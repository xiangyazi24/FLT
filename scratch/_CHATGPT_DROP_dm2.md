The main trick is to prove the recursive body-map lemma once.  Because the body argument is dependent in the proof `m < n`, the body lemma quantifies over **both** proof arguments:

```lean
∀ m (hmS hmR : m < n), uS m hmS = φ (uR m hmR)
```

That makes the `simp [hu, map_sum]` proof insensitive to the specific proof witnesses created inside the `WellFounded.fix` body.

```lean
import Mathlib.AlgebraicGeometry.EllipticCurve.Projective.Formula
import Mathlib.RingTheory.PowerSeries.Substitution
import Mathlib.RingTheory.MvPowerSeries.Rename
import Mathlib.Tactic

open scoped BigOperators
open MvPowerSeries

namespace WeierstrassCurve

noncomputable section

variable {R S : Type*} [CommRing R] [CommRing S]

/--
Naturality of the recursive body defining `formalUCoeff`.

This is the key lemma.  The important point is that `uS` and `uR` are dependent in
proofs of `< n`, so the compatibility hypothesis must quantify over both proof
witnesses.  Then `simp [hu]` can rewrite all recursive calls inside the sums,
regardless of which arithmetic proof was generated by `omega` in the definition.
-/
private lemma formalUCoeffBody_map
    (φ : R →+* S) (W : WeierstrassCurve R) :
    ∀ (n : ℕ)
      (uS : (m : ℕ) → m < n → S)
      (uR : (m : ℕ) → m < n → R),
      (∀ m (hmS : m < n) (hmR : m < n),
        uS m hmS = φ (uR m hmR)) →
        formalUCoeffBody (W.map φ) n uS
          = φ (formalUCoeffBody W n uR) := by
  classical
  intro n
  cases n with
  | zero =>
      intro uS uR hu
      simp [formalUCoeffBody, WeierstrassCurve.map]
  | succ n =>
      intro uS uR hu
      -- The recursive body has conditionals at `n ≥ 1`, `n ≥ 2`,
      -- `n ≥ 3`, and `n ≥ 5`.  Splitting them first gives both sides
      -- the same branches and makes `simp` deterministic.
      by_cases h₁ : n ≥ 1 <;>
      by_cases h₂ : n ≥ 2 <;>
      by_cases h₃ : n ≥ 3 <;>
      by_cases h₅ : n ≥ 5 <;>
        simp [formalUCoeffBody, WeierstrassCurve.map,
          h₁, h₂, h₃, h₅,
          hu, map_sum, RingHom.map_add, RingHom.map_mul]

/--
Coefficient naturality for the recursive `U`-series coefficients.
-/
theorem formalUCoeff_map
    (φ : R →+* S) (W : WeierstrassCurve R) (n : ℕ) :
    formalUCoeff (W.map φ) n = φ (formalUCoeff W n) := by
  classical
  refine Nat.strongRecOn n ?_
  intro n ih
  rw [formalUCoeff_eq (W.map φ) n, formalUCoeff_eq W n]
  exact formalUCoeffBody_map φ W n
    (fun m _ => formalUCoeff (W.map φ) m)
    (fun m _ => formalUCoeff W m)
    (by
      intro m hmS hmR
      exact ih m hmS)

/--
Power-series naturality of `formalU`.
-/
theorem formalU_map
    (φ : R →+* S) (W : WeierstrassCurve R) :
    PowerSeries.map φ (formalU W) = formalU (W.map φ) := by
  ext n
  simp [formalU, formalUCoeff_map φ W n]

/--
Power-series naturality of `formalW = X^3 * formalU`.
-/
theorem formalW_map
    (φ : R →+* S) (W : WeierstrassCurve R) :
    PowerSeries.map φ (formalW W) = formalW (W.map φ) := by
  simp [formalW, formalU_map φ W]

/--
The coefficient curve over `MvPowerSeries (Fin 2)` commutes with mapping
coefficients by `φ`.
-/
private lemma formalCoeffCurve_map
    (φ : R →+* S) (W : WeierstrassCurve R) :
    (((W.map (MvPowerSeries.C : R →+* MvPowerSeries (Fin 2) R)).toProjective).map
        (MvPowerSeries.map φ))
      =
    (((W.map φ).map
        (MvPowerSeries.C : S →+* MvPowerSeries (Fin 2) S)).toProjective) := by
  ext <;> simp [WeierstrassCurve.map]

/--
Naturality of the formal point map.

The `z` coordinate is the only nontrivial case.  It uses
`PowerSeries.map_subst`, with the substitution point `MvPowerSeries.X i`.
-/
theorem formalPointMv_map_comp
    (φ : R →+* S) (W : WeierstrassCurve R) (i : Fin 2) :
    (MvPowerSeries.map φ) ∘ formalPointMv W i
      = formalPointMv (W.map φ) i := by
  classical
  funext j
  fin_cases j
  · simp [formalPointMv]
  · simp [formalPointMv]
  · -- `PowerSeries.map_subst` rewrites
    --   map φ ((formalW W).subst (X i))
    -- to
    --   (map φ (formalW W)).subst (map φ (X i)).
    rw [show
        ((MvPowerSeries.map φ) ((formalPointMv W i) 2))
          = ((PowerSeries.map φ (formalW W)).subst
              (MvPowerSeries.map φ (MvPowerSeries.X i))) by
          simp [formalPointMv,
            PowerSeries.map_subst (PowerSeries.HasSubst.X i)
              (h := φ) (formalW W)]]
    simp [formalPointMv, formalW_map φ W]

/--
Function-level naturality of `formalAddXYZ`.
-/
theorem formalAddXYZ_map_comp
    (φ : R →+* S) (W : WeierstrassCurve R) :
    (MvPowerSeries.map φ) ∘ formalAddXYZ W
      = formalAddXYZ (W.map φ) := by
  classical
  let C_R : R →+* MvPowerSeries (Fin 2) R := MvPowerSeries.C
  let C_S : S →+* MvPowerSeries (Fin 2) S := MvPowerSeries.C
  let Φ : MvPowerSeries (Fin 2) R →+* MvPowerSeries (Fin 2) S :=
    MvPowerSeries.map φ

  let P : Fin 3 → MvPowerSeries (Fin 2) R := formalPointMv W (0 : Fin 2)
  let Q : Fin 3 → MvPowerSeries (Fin 2) R := formalPointMv W (1 : Fin 2)
  let P' : Fin 3 → MvPowerSeries (Fin 2) S := formalPointMv (W.map φ) (0 : Fin 2)
  let Q' : Fin 3 → MvPowerSeries (Fin 2) S := formalPointMv (W.map φ) (1 : Fin 2)

  let W₀ : WeierstrassCurve.Projective (MvPowerSeries (Fin 2) R) :=
    (W.map C_R).toProjective
  let W₁ : WeierstrassCurve.Projective (MvPowerSeries (Fin 2) S) :=
    ((W.map φ).map C_S).toProjective

  have hcurve : W₀.map Φ = W₁ := by
    dsimp [W₀, W₁, C_R, C_S, Φ]
    exact formalCoeffCurve_map φ W

  have hP : Φ ∘ P = P' := by
    dsimp [P, P', Φ]
    exact formalPointMv_map_comp φ W (0 : Fin 2)

  have hQ : Φ ∘ Q = Q' := by
    dsimp [Q, Q', Φ]
    exact formalPointMv_map_comp φ W (1 : Fin 2)

  have hXYZ :
      (W₀.map Φ).addXYZ (Φ ∘ P) (Φ ∘ Q)
        = Φ ∘ W₀.addXYZ P Q := by
    exact WeierstrassCurve.Projective.map_addXYZ
      (W' := W₀) (f := Φ) (P := P) (Q := Q)

  funext j
  calc
    ((MvPowerSeries.map φ) ∘ formalAddXYZ W) j
        = Φ ((W₀.addXYZ P Q) j) := by
            simp [formalAddXYZ, W₀, P, Q, C_R, Φ]
    _ = ((W₀.map Φ).addXYZ (Φ ∘ P) (Φ ∘ Q)) j := by
            exact (congrFun hXYZ j).symm
    _ = (W₁.addXYZ P' Q') j := by
            simp [hcurve, hP, hQ]
    _ = formalAddXYZ (W.map φ) j := by
            simp [formalAddXYZ, W₁, P', Q', C_S]

/--
Pointwise naturality of `formalAddXYZ`.
-/
theorem formalAddXYZ_map
    (φ : R →+* S) (W : WeierstrassCurve R) (j : Fin 3) :
    MvPowerSeries.map φ (formalAddXYZ W j)
      = formalAddXYZ (W.map φ) j := by
  exact congrFun (formalAddXYZ_map_comp φ W) j

/-- Optional coordinate wrapper: the formal `X` coordinate. -/
theorem formalAddX_map
    (φ : R →+* S) (W : WeierstrassCurve R) :
    MvPowerSeries.map φ (formalAddXYZ W (0 : Fin 3))
      = formalAddXYZ (W.map φ) (0 : Fin 3) := by
  exact formalAddXYZ_map φ W (0 : Fin 3)

/-- Optional coordinate wrapper: the formal `Y` coordinate. -/
theorem formalAddY_map
    (φ : R →+* S) (W : WeierstrassCurve R) :
    MvPowerSeries.map φ (formalAddXYZ W (1 : Fin 3))
      = formalAddXYZ (W.map φ) (1 : Fin 3) := by
  exact formalAddXYZ_map φ W (1 : Fin 3)

/-- Optional coordinate wrapper: the formal `Z` coordinate. -/
theorem formalAddZ_map
    (φ : R →+* S) (W : WeierstrassCurve R) :
    MvPowerSeries.map φ (formalAddXYZ W (2 : Fin 3))
      = formalAddXYZ (W.map φ) (2 : Fin 3) := by
  exact formalAddXYZ_map φ W (2 : Fin 3)

end

end WeierstrassCurve
```

## If the body lemma needs one extra nudge

If your local `formalUCoeffBody` has explicit proof terms that `simp [hu]` does not rewrite automatically inside the attached sums, replace the last line of the successor case in `formalUCoeffBody_map` by this slightly more aggressive version:

```lean
        simp only [formalUCoeffBody, WeierstrassCurve.map,
          h₁, h₂, h₃, h₅,
          RingHom.map_zero, RingHom.map_one,
          RingHom.map_add, RingHom.map_mul,
          map_sum]
        simp [hu]
```

The reason this works is that after `map_sum` and `RingHom.map_mul`, every recursive occurrence has the form

```lean
uS m hmS = φ (uR m hmR)
```

for some proof witnesses `hmS hmR : m < n + 1`, and the `hu` hypothesis was deliberately stated with both witnesses explicit.
