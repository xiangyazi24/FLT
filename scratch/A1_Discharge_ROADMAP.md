```lean
open WeierstrassCurve.Affine

local notation "ℚbar" => AlgebraicClosure ℚ

abbrev RatPts (E : WeierstrassCurve ℚ) [E.IsElliptic] : Type :=
  (E⁄ℚ).Point

abbrev RatTors (E : WeierstrassCurve ℚ) [E.IsElliptic] : Type :=
  AddCommGroup.torsion (RatPts E)

abbrev GeomPts (E : WeierstrassCurve ℚ) [E.IsElliptic] : Type :=
  (E⁄ℚbar).Point

abbrev GeomNTorsion
    (E : WeierstrassCurve ℚ) [E.IsElliptic] (N : ℕ) : Type :=
  Submodule.torsionBy ℤ (GeomPts E) N

noncomputable abbrev ratToQbarAlgHom : ℚ →ₐ[ℚ] ℚbar :=
  Algebra.ofId ℚ ℚbar
```

Now define the map

\[
E(\mathbb Q)_{\mathrm{tors}}\to E[N](\overline{\mathbb Q})
\]

where `N = Nat.card (RatTors E)`. I parameterize it by the killing lemma so the definition is reusable.

```lean
noncomputable def ratTorsToGeomNTorsion
    (E : WeierstrassCurve ℚ) [E.IsElliptic]
    (N : ℕ)
    (hkill : ∀ T : RatTors E, N • T = 0) :
    RatTors E →+ GeomNTorsion E N where
  toFun T :=
    ⟨ratPointBaseChange E (T : RatPts E), by
      have hT : N • (T : RatPts E) = 0 := by
        simpa using congrArg Subtype.val (hkill T)
      have hmap := congrArg (ratPointBaseChange E) hT
      simpa [GeomNTorsion, ratPointBaseChange] using hmap⟩
  map_zero' := by
    ext
    simp [ratPointBaseChange]
  map_add' := by
    intro P Q
    ext
    simp [ratPointBaseChange]

lemma ratTorsToGeomNTorsion_injective
    (E : WeierstrassCurve ℚ) [E.IsElliptic]
    (N : ℕ)
    (hkill : ∀ T : RatTors E, N • T = 0) :
    Function.Injective (ratTorsToGeomNTorsion E N hkill) := by
  classical
  intro P Q hPQ
  apply Subtype.ext
  apply ratPointBaseChange_injective E
  exact congrArg Subtype.val hPQ
```

```lean
lemma addOrderOf_ratTors_coe
```

```lean
lemma hasRationalPointOfOrder_of_ratTors_equiv
    (E : WeierstrassCurve ℚ) [E.IsElliptic]
    {m n : ℕ}
    (hn : 0 < n)
    (e : RatTors E ≃+ ZMod m × ZMod n) :
    HasRationalPointOfOrder E n := by
  let z : ZMod m × ZMod n := (0, 1)
  let T : RatTors E := e.symm z

  have hz : addOrderOf z = n := by
    rw [Prod.addOrderOf]
    simp [z, ZMod.addOrderOf_one, hn.ne']

  have hT_sub : addOrderOf T = n := by
    simpa [T, hz] using
      (e.symm.addOrderOf_eq z)

  have hT_pts : addOrderOf ((T : RatPts E)) = n := by
    rw [addOrderOf_ratTors_coe E T, hT_sub]

  --   ∃ P : (E⁄ℚ).Point, addOrderOf P = n,
  -- this is exactly:

    (show ∃ P : RatPts E, addOrderOf P = n from
```

```lean
  have hkill : ∀ T : G, N • T = 0 := by

      has_point_order_n := ?_
```
